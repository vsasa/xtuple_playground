function hasData()
{
    var cparam = new Object();
    var data = toolbox.executeQuery('SELECT count(*) as count FROM xthelloworld.fin_suban', cparam);
    if (data.first())
    {
        if (data.value("count") > 0)
        {
            return true;
        } else {
            return false;
        }
    }
};
